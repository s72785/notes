		int fileenvigenere(char* filename, int *a, int l) {
			FILE* ifp;
			ifp = fopen(filename, "r");
			char c='1';
			int i=0;
			if (ifp == NULL) file_error();
			while(fscanf(ifp, "%c", &c) != EOF) {
				i++;
				if(i==l) {
					i=0;
				}else{
					i++;
				}
				printf("%c", encaesar( c, a[i] ));
			}
			fclose(ifp);
			printf("\n");
			return 0;
		}
